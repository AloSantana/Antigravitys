    Network/Server Issue
         │
         ▼
    ws.onclose()
         │
         ├─► reconnectAttempts++
         ├─► updateConnectionStatus(false)
         │   └─► UI: [●] Reconnecting... (attempt 1)
         │
         ▼
    calculateReconnectDelay()
         │
         ├─► exponentialDelay = 1000 × (1.5 ^ attempts)
         ├─► cappedDelay = min(exponentialDelay, 30000)
         ├─► jitter = random(-0.2, 0.2)
         └─► finalDelay = cappedDelay × (1 + jitter)
         │
         ▼
    setTimeout(connectWebSocket, finalDelay)
         │
         ▼
    ┌──────── Wait for Delay ────────┐
